	js 이벤트핸들러 함수 속성
	onblur = function () { ... };

	** 이벤트 종류: 이벤트 함수의 배개변수에서 이벤트 type을 넣는 매개변수 값을 의미(문자열로 기술)
	이벤트 종류: "blur"

	js
	- 객체나 엘리먼트명.addEventListener('이벤트종류', function () { ... });

	jquery
	- $(객체명).on('이벤트종류', function () { ... });
	- $(객체명).bind('이벤트종류', function () { ... });
